```python
def count_collisions(d, n, G):
    table = {}
    collisions = 0
    for i in range(n):
        for j in range(n):
            point = (i * d + j) * G
            key = (point.x, point.y)
            if key in table: 
                collisions += 1
            else:
                table[key] = True
    return collisions
```
